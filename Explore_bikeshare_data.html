#Most Common year of birth
# Create the function.
getmode <- function(v) {
   uniqv <- unique(v)
   uniqv[which.max(tabulate(match(v, uniqv)))]
}

# Calculate the mode using the user function.
result <- getmode(ny$Birth.Year[!is.na(ny$Birth.Year)])
print(result)
